建立或編輯 `/etc/sysctl.d/99-custom.conf`
```ini
# 禁止快捷鍵操作低階指令
kernel.sysrq = 0

# TCP 增強
# 忽略錯誤且不合理的 ICMP 錯誤訊息
net.ipv4.icmp_ignore_bogus_error_responses = 1
# 防止 IP 欺騙攻擊，系統會檢查收到封包的來源地址是否符合路由表的合理回應
net.ipv4.conf.default.rp_filter = 1
net.ipv4.conf.all.rp_filter = 1
# 不接受 IP 封包中的 source routing 選項
net.ipv4.conf.all.accept_source_route = 0
net.ipv6.conf.all.accept_source_route = 0
# 不發送 ICMP Redirect 封包給其他裝置 (非路由器)
net.ipv4.conf.all.send_redirects = 0
net.ipv4.conf.default.send_redirects = 0
# 不接受 ICMP Redirect 封包 (非路由器)
net.ipv4.conf.all.accept_redirects = 0
net.ipv4.conf.default.accept_redirects = 0
net.ipv4.conf.all.secure_redirects = 0
net.ipv4.conf.default.secure_redirects = 0
net.ipv6.conf.all.accept_redirects = 0
net.ipv6.conf.default.accept_redirects = 0

# TCP 防禦
# 啟用 SYN Cookies 防止 SYN Flood 攻擊
net.ipv4.tcp_syncookies = 1
# 防止 TIME-WAIT 攻擊（又稱 TCP 端口復用漏洞）
net.ipv4.tcp_rfc1337 = 1

# TCP 優化
# 加速 TCP 連線建立過程
net.ipv4.tcp_fastopen = 3
# tcp  擁塞控制算法，能提高網路吞吐量和降低延遲
net.ipv4.tcp_congestion_control = cubic
# 封包排隊演算法，影響延遲
net.core.default_qdisc = fq_codel
```

啓用：
```shell
sudo sysctl --system # 讀取 sysctl.d/ 所有設定
```
